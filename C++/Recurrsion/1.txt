// Function calls itself 

#include <bits/stdc++.h>
using namespace std;

void f(String s , int n){
    for(int i = 0 ; i < 0 ; i++){
        cout<<s;
        if(i == n){
            return;
        }
    }
}

int main(){
    String s;
    cin>>s;

    int n;
    cin>>n;

    f(s,n)<<endl;

    return 0;
}
